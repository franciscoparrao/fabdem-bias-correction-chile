% =============== Inputs (left column) ===============
\node[data] (fabdem) {FABDEM v1.2\\\scriptsize via GEE};
\node[data, below=of fabdem] (sentinel) {Sentinel-1/2\\\scriptsize Planetary Computer};
\node[data, below=of sentinel] (atl08) {ICESat-2 ATL08\\\scriptsize earthaccess};
\node[data, below=of atl08] (egm08) {EGM2008 geoid\\\scriptsize PROJ network};

% =============== Per-tile feature derivation ===============
\node[feature, right=14mm of fabdem] (terrain) {SurtGis terrain\\\scriptsize slope, openness, MRVBF\ldots};
\node[feature, below=of terrain] (hydro) {SurtGis hydrology\\\scriptsize HAND, TWI, flow\ldots};
\node[feature, below=of hydro] (s2) {Sentinel-2 indices\\\scriptsize NDVI, NDWI, NDMI\ldots};
\node[feature, below=of s2] (s1) {Sentinel-1 SAR\\\scriptsize VV, VH, ratio};

% =============== Footprint pipeline ===============
\node[process, right=14mm of terrain] (extract) {Extract ATL08\\\scriptsize quality filter, EGM2008\\\scriptsize WGS84 \(\to\) orthometric};
\node[process, below=18mm of extract] (sample) {Sample features\\\scriptsize at footprint coords\\\scriptsize (CRS-aware nearest)};
\node[process, below=of sample] (target) {Build regression target\\\scriptsize \(y = h_{te,\text{ortho}} - h_{\text{fabdem}}\)};

% =============== ML training ===============
\node[model, right=14mm of sample] (train) {XGBoost + Optuna\\\scriptsize 100 trials, spatial-block CV\\\scriptsize 10 km blocks, K=5};

% =============== Outputs ===============
\node[output, right=14mm of train, yshift=10mm] (rmse) {OOF RMSE\\\scriptsize 3.054 \(\to\) 2.483 m};
\node[output, right=14mm of train, yshift=-10mm] (cog) {Corrected DEM\\\scriptsize Cloud-Optimized GeoTIFF};
\node[output, below=of cog] (ood) {OOD evaluation\\\scriptsize 8 climate regimes\\\scriptsize 16 tiles, 181k footprints};
